22  		
23  		  String driverPath="";

30  		option.addArguments("--disable-notifications");
31  		if (System.getProperty("os.name").toUpperCase().contains("MAC"))
32  	       option.addArguments("-start-fullscreen");
33  	       else if (System.getProperty("os.name").toUpperCase().contains("WINDOWS"))
34  		option.addArguments("--start-maximazed");
35  		else throw new IllegalArgumentException("UknowOS");
36  		
37  		driver = new ChromeDriver(option);
38  		 wait = new WebDriverWait (driver,15);
39  		
40  	}
41  	else if (Browsers.mybrowser.equals("Firefox")) {
42  		
43  		
44

48  		       String driverPath="";

54  			 wait = new WebDriverWait (driver,15);
55  		
56  	}
57  	
58  	
59

64

66

68  		driver.manage().window().maximize();
69  		 wait = new WebDriverWait (driver,15);

71  }
